# Design Token System

---

## Overview

User choices in the Design System step map to a complete set of CSS custom properties and Tailwind-compatible JSON. Token derivation is a **pure function** — not stored in Zustand, computed on the fly.

---

## Derivation Function

```typescript
// src/tokens/derive.ts

export interface DerivedTokens {
  colors: {
    accent: Record<'50'|'100'|'200'|'300'|'400'|'500'|'600'|'700'|'800'|'900', string>;
  };
  typography: {
    fontFamily: string;
    sizes: Record<'xs'|'sm'|'base'|'lg'|'xl'|'2xl'|'3xl', string>;
  };
  spacing: Record<'1'|'2'|'3'|'4'|'6'|'8'|'12'|'16'|'24', string>;
  radius: Record<'sm'|'md'|'lg'|'xl'|'full', string>;
  shadow: Record<'sm'|'md'|'lg', string>;
}

export function deriveTokens(designSystem: DesignSystemData): DerivedTokens {
  return {
    colors: {
      accent: generateAccentScale(designSystem.accentColorHex),
    },
    typography: {
      fontFamily: fontFamilyMap[designSystem.fontFamily],
      sizes: defaultTypographySizes, // fixed scale, not user-configurable in MVP
    },
    spacing: generateSpacingScale(designSystem.spacingBase),
    radius: radiusScaleMap[designSystem.radiusScale],
    shadow: shadowMap[designSystem.shadowDepth],
  };
}
```

## Accent Color Scale Generation

No external color library. Hand-rolled HSL math (~30 lines):

```typescript
// src/tokens/colorScale.ts

function hexToHsl(hex: string): [number, number, number] {
  // Convert #rrggbb to [h, s, l] in degrees/percent
  const r = parseInt(hex.slice(1, 3), 16) / 255;
  const g = parseInt(hex.slice(3, 5), 16) / 255;
  const b = parseInt(hex.slice(5, 7), 16) / 255;
  const max = Math.max(r, g, b), min = Math.min(r, g, b);
  const l = (max + min) / 2;
  const d = max - min;
  const s = d === 0 ? 0 : d / (1 - Math.abs(2 * l - 1));
  const h = d === 0 ? 0 : max === r ? ((g - b) / d) % 6
    : max === g ? (b - r) / d + 2
    : (r - g) / d + 4;
  return [Math.round(h * 60 + 360) % 360, Math.round(s * 100), Math.round(l * 100)];
}

function hslToHex(h: number, s: number, l: number): string {
  // Convert HSL to #rrggbb
  s /= 100; l /= 100;
  const k = (n: number) => (n + h / 30) % 12;
  const a = s * Math.min(l, 1 - l);
  const f = (n: number) => l - a * Math.max(-1, Math.min(k(n) - 3, Math.min(9 - k(n), 1)));
  return '#' + [f(0), f(8), f(4)].map((x) => Math.round(x * 255).toString(16).padStart(2, '0')).join('');
}

const LIGHTNESS_SCALE = { 50: 96, 100: 92, 200: 82, 300: 70, 400: 58,
                          500: 48, 600: 40, 700: 32, 800: 24, 900: 15 };

export function generateAccentScale(hex: string): Record<string, string> {
  const [h, s] = hexToHsl(hex);
  const cappedS = Math.min(s, 85); // prevent blown-out saturations
  return Object.fromEntries(
    Object.entries(LIGHTNESS_SCALE).map(([key, l]) => [key, hslToHex(h, cappedS, l)])
  );
}
```

Test with: blue (`#3b82f6`), rose (`#f43f5e`), emerald (`#10b981`), purple (`#8b5cf6`), neutral gray (`#6b7280`).

## Spacing Scale

```typescript
function generateSpacingScale(base: 4 | 8 | 12 | 16): Record<string, string> {
  const b = base / 16; // convert px to rem
  return {
    '1': `${b * 0.25}rem`,
    '2': `${b * 0.5}rem`,
    '3': `${b * 0.75}rem`,
    '4': `${b}rem`,
    '6': `${b * 1.5}rem`,
    '8': `${b * 2}rem`,
    '12': `${b * 3}rem`,
    '16': `${b * 4}rem`,
    '24': `${b * 6}rem`,
  };
}
```

## tokens.tailwind.json Output

Flat key-value pairs matching Tailwind 4 `@theme` variable names.

```json
{
  "--color-accent-50": "#eff6ff",
  "--color-accent-500": "#3b82f6",
  "--color-accent-900": "#1e3a8a",
  "--font-sans": "'Geist', system-ui, sans-serif",
  "--radius-md": "0.375rem",
  "--spacing-4": "1rem",
  "--shadow-md": "0 4px 6px -1px rgb(0 0 0 / 0.1)"
}
```

User drops this into their project and uses in `globals.css`:

```css
@import "tailwindcss";
@theme inline {
  /* Paste token values here, or @import the JSON (Tailwind 4 feature) */
  --color-accent-500: #3b82f6;
}
```

## Preview Step

`PreviewStep` applies derived tokens as inline `style` prop on a container div:

```tsx
const tokens = deriveTokens(designSystem);
const cssVars = {
  '--preview-accent': tokens.colors.accent['500'],
  '--preview-accent-light': tokens.colors.accent['100'],
  '--preview-font': tokens.typography.fontFamily,
  '--preview-radius': tokens.radius.md,
  // ...
};

return (
  <div style={cssVars as React.CSSProperties}>
    {/* Color swatches, typography sample, spacing grid */}
  </div>
);
```

This never touches the wizard's own Tailwind theme. The `--preview-*` prefix avoids conflicts.
